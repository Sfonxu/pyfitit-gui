461    def expand_deformations(self):
462        """Fucntion that translates a list of deformations into a string
463        representation that would be encountered in a PyFitIt project file"""
464        # pylint: disable=line-too-long
465        deform_string_list = []
466        for deformation in self.deformations:
467            if deformation.def_type == "shift":
468                deform_string_list.append(f'    deformation = "{deformation.name}"\n')
469                deform_string_list.append(
470                    f"    axis = normalize(m.atom[{deformation.atom_1}]-m.atom[{deformation.atom_2}])\n"
471                )
472                deform_string_list.append(
473                    f"    m.part[{deformation.part}].shift(axis*params[deformation])\n\n"
474                )
475            elif deformation.def_type == "rotation":
476                deform_string_list.append(f'    deformation = "{deformation.name}"\n')
477                deform_string_list.append(
478                    f"    axis = normalize(m.atom[{deformation.atom_1}]-m.atom[{deformation.atom_2}])\n"
479                )
480                deform_string_list.append(
481                    f"    m.part[{deformation.part}].rotate(axis, m.atom[{deformation.atom_1}], params[deformation])\n\n"
482                )
483
484        if deform_string_list:
485            deformation_string = "".join(deform_string_list)
486            return deformation_string
487        self.save_and_exit_error_message(
488            "No deformations defined, unable to generate project!"
489        )
490        return ""

462    def expand_deformations(self):
463        """Fucntion that translates a list of deformations into a string
464        representation that would be encountered in a PyFitIt project file"""
465        # pylint: disable=line-too-long
466        deform_string_list = []
467        for deformation in self.deformations:
468            if deformation.def_type == "shift":
469                deform_string_list.append(f'    deformation = "{deformation.name}"\n')
470                deform_string_list.append(
471                    f"    axis = normalize(m.atom[{deformation.atom_1}]-m.atom[{deformation.atom_2}])\n"
472                )
473                deform_string_list.append(
474                    f"    m.part[{deformation.part}].shift(axis*params[deformation])\n\n"
475                )
476            elif deformation.def_type == "rotation":
477                deform_string_list.append(f'    deformation = "{deformation.name}"\n')
478                deform_string_list.append(
479                    f"    axis = normalize(m.atom[{deformation.atom_1}]-m.atom[{deformation.atom_2}])\n"
480                )
481                deform_string_list.append(
482                    f"    m.part[{deformation.part}].rotate(axis, m.atom[{deformation.atom_1}], params[deformation])\n\n"
483                )
484
485        if deform_string_list:
486            deformation_string = "".join(deform_string_list)
487            return deformation_string
488        self.save_and_exit_error_message(
489            "No deformations defined, unable to generate project!"
490        )
491        return ""

def expand_deformations(self): 
462    def expand_deformations(self):
463        """Fucntion that translates a list of deformations into a string
464        representation that would be encountered in a PyFitIt project file"""
465        # pylint: disable=line-too-long
466        deform_string_list = []
467        for deformation in self.deformations:
468            if deformation.def_type == "shift":
469                deform_string_list.append(f'    deformation = "{deformation.name}"\n')
470                deform_string_list.append(
471                    f"    axis = normalize(m.atom[{deformation.atom_1}]-m.atom[{deformation.atom_2}])\n"
472                )
473                deform_string_list.append(
474                    f"    m.part[{deformation.part}].shift(axis*params[deformation])\n\n"
475                )
476            elif deformation.def_type == "rotation":
477                deform_string_list.append(f'    deformation = "{deformation.name}"\n')
478                deform_string_list.append(
479                    f"    axis = normalize(m.atom[{deformation.atom_1}]-m.atom[{deformation.atom_2}])\n"
480                )
481                deform_string_list.append(
482                    f"    m.part[{deformation.part}].rotate(axis, m.atom[{deformation.atom_1}], params[deformation])\n\n"
483                )
484
485        if deform_string_list:
486            deformation_string = "".join(deform_string_list)
487            return deformation_string
488        self.save_and_exit_error_message(
489            "No deformations defined, unable to generate project!"
490        )
491        return ""
